```tsx
import React from "react"

import Card from "@Components/Card"
import DoubleSlider from "@Components/DoubleSlider"
import SingleSlider from "@Components/SingleSlider"
import DoubleSlider from "@Components/DoubleSliderWithPercentage"
import Syntax from "@Components/Syntax"

let sliderVal: number[] = [0, 1000]
let selected: number[] = [300, 700]

const exampleCode = `
<SingleSlider
    update={this.sliderChange}
    range={[0, 999]}
    value={this.state.slider}
    name={"slider"}
/>
`.trim()

const exampleDoubleCode = `
<DoubleSlider
    range={[0, 1000]}
    selection={this.state.select}
    updateValue={this.updateSlider}
    name={"doubleSlider"}
/>
    `.trim()

const exampleDoubleSmallCode = `
<DoubleSlider
    range={[0, 1000]}
    selection={this.state.select}
    updateValue={this.updateSlider}
    name={"smallSlider"}
    small
/>
    `.trim()

type SlidePageState = {
    slider: number
    doubleSlider: number[]
    doubleSmallSlider: number[]
}

class SliderPage extends React.Component<{}, SlidePageState> {
    constructor(props: any) {
        super(props)
        this.state = {
            slider: 0,
            doubleSilder: [...selected],
            doubleSmallSlider: [...selected],
        }
    }

    updateDoubleSlider = (name: string, value: number[]) => {
        this.setState({ doubleSlider: value })
    }

    updateDoubleSmallSlider = (name: string, value: number[]) => {
        this.setState({ doublesSmallSlider: value })
    }

    sliderChange = (name: string, value: number) => {
        this.setState({ slider: value })
    }

    render() {
        return (
            <Card>
                <Syntax>{`import Slider from "@Components/SingleSlider"`}</Syntax>
            </Card>
            <Card>
                <SingleSlider
                    update={this.sliderChange}
                    range={[0, 999]}
                    value={this.state.slider}
                    name={"slider"}
                />
                    <div>{this.state.slider}</div>
                    <Syntax>{exampleCode}</Syntax>
            </Card>
            <Card>
                <Syntax>{`import DoubleSlider from "@Components//DoubleSlider"`}</Syntax>
            </Card>
            <Card>
                <DoubleSlider
                    range={sliderVal}
                    value={this.state.doubleSlider}
                    update={this.updateDoubleSlider}
                    name={"doubleSlider"}
                />
                <div>{this.state.doubleSlider.join(", ")}</div>
                <Syntax>{exampleDoubleCode}</Syntax>
            </Card>
            <Card>
                <DoubleSlider
                    range={sliderVal}
                    value={this.state.doubleSmallSlider}
                    update={this.updateDoubleSmallSlider}
                    name={"smallSlider"}
                    small
                />
                <div>{this.state.doubleSmallSlider.join(", ")}</div>
                <Syntax>{exampleDoubleSmallCode}</Syntax>
            </Card>
        )
    }
}

export default SliderPage
```
